```
git clone http #你要克隆的地址
git config --global user.email "你的邮箱"
git config --global user.user "你的github名字" 
git add . 					  #添加到本地仓库, 修改之后也要add
git status #查询仓库当前状态

git commit -m "注释" # 本地仓库存档, 存档之后才可以提交到远程仓库

git pull # 提交之前要pull
git push #提交到远程仓库

# git remote add  远程仓库名 远程仓库的URL    # 如果需要就写
```
